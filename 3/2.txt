#include <iostream>
#include <stdlib.h>
#include <time.h>

using namespace std;

int col_srz(int m[], int R, int C, int n) {
    float srz = 0;
    int kol = 0;
    int summ = 0;
    if (n < C/2){
        for (int i = 0; i < R/2; i++) {
        summ += m[i*R + n]; 
        kol ++;
        }
    }
    else if (n < C and n >= C/2) {
        for (int i = R / 2; i < R; i++) {
        summ += m[i*R + n]; 
        kol ++;
            
        }
    }
    
    srz = summ / kol;
    return srz;
}

#define N 8
#define M 10

int main()
{
    int matrix[N*M];
    srand(time(NULL));
    
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < M; j++) {
            matrix[i*N + j] = rand()%100;
            cout << matrix[i*N + j] << "\t";
        }
        cout << endl;
    }
    cout << endl;
    
    int srz1;
    for (int i =  0; i < M; i++) {
        srz1 = col_srz(matrix, N, M, i);
        cout << "Column " << i+1 << " - " << srz1 << endl;
    }
    
    return 0;
}
