#include <iostream>
#include <stdlib.h>
#include <time.h>

using namespace std;

#define SIZE 10

int M[SIZE][SIZE];
int array[SIZE];
int zeros = 0;
int pluses = 0;

int main()

{
    srand(time(NULL));

    cout << "Matrix M["<< SIZE<< "][" << SIZE << "]:" << endl;

    for(int i = 0; i < SIZE; i ++) {
       for(int j = 0; j < SIZE; j ++) {
           M[i][j] = rand()%201 - 100;
           cout << (M [i][j]) <<  "\t";
       }
       cout << endl;
   }
   
   for (int i = 0; i < SIZE/2; i++) {
       for (int j = i; j < SIZE - i; j++) {
           if (M[i][j] == 0) {
               zeros++;
           }
       }
   }
   
   for (int i = 5; i < SIZE; i++) {
       for (int j = SIZE - i - 1; j <= i; j++) {
           if (M[i][j] > 0) {
               pluses++;
           }
       }
   }
   
   for (int j = 0; j < SIZE; j++) {
       int min = M[0][j];
       for (int i = 1; i < SIZE; i++) {
           if ( abs(M[i][j]) < abs(min) ) {
               min = M[i][j];
           }
       }
       array[j] = min;
   }
   
   cout << "ZEROS: " << zeros << endl << "PLUSES: " << pluses << endl;
   
   cout << "ARRAY OF MIN: ";
   for (int i = 0; i < SIZE; i++) {
       cout << array[i] << " ";
   }
   
   return 0;

}
