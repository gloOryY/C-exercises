#include <stdio.h>
int main()
{
    for(int d = 1; d < 1001; d++){
        if(d == 1){
            printf("%d рубль\n",d);
        }
        if(2 <= d && d <= 4){
             printf("%d рубля\n",d);
        }
        if(5 <= d && d <= 20){
             printf("%d рублей\n",d);
        }
        if(d > 9 && d < 100 && d % 10 == 1 && d != 11){
             printf("%d рублей\n",d);
        }
        if(d > 99 && d < 1000 && d % 10 == 1 && d%100 != 11){
             printf("%d рубль\n",d);
        }
        if(d > 99 && d < 1000 && 5 <= d%100 && d%100 <= 20){
             printf("%d рублей\n",d);
        }
        if(d > 99 && d < 1000 && 2 <= d%100 && d%100 <= 4){
            printf("%d рублей\n",d);
        }
    }
    return 0;
}
